# RADICAL SEA BUNNY - Dockerfile
# Based on Alpine Linux for minimal footprint

FROM python:3.10-alpine

# Metadata
LABEL maintainer="Security Research Team"
LABEL version="2.0.0"
LABEL description="RADICAL SEA BUNNY - Ultimate Cybersecurity Command & Control Platform"

# Install system dependencies
RUN apk add --no-cache \
    # Build dependencies
    gcc \
    musl-dev \
    libffi-dev \
    openssl-dev \
    python3-dev \
    # Network tools
    nmap \
    curl \
    netcat-openbsd \
    iputils \
    traceroute \
    bind-tools \
    whois \
    openssh-client \
    # Additional tools
    nmap-nping \
    tcpdump \
    iptables \
    # Fonts for matplotlib
    ttf-dejavu \
    ttf-droid \
    ttf-freefont \
    # Utilities
    bash \
    vim \
    jq \
    && rm -rf /var/cache/apk/*

# Install nikto
RUN apk add --no-cache nikto || \
    (wget -q -O /tmp/nikto.tar.gz https://github.com/sullo/nikto/archive/master.tar.gz && \
     tar -xzf /tmp/nikto.tar.gz -C /opt/ && \
     ln -s /opt/nikto-master/program/nikto.pl /usr/local/bin/nikto && \
     chmod +x /usr/local/bin/nikto && \
     rm /tmp/nikto.tar.gz)

# Install signal-cli (if available)
RUN wget -q -O /tmp/signal-cli.tar.gz https://github.com/AsamK/signal-cli/releases/download/v0.11.8/signal-cli-0.11.8.tar.gz && \
    tar -xzf /tmp/signal-cli.tar.gz -C /opt/ && \
    ln -s /opt/signal-cli-0.11.8/bin/signal-cli /usr/local/bin/signal-cli && \
    chmod +x /usr/local/bin/signal-cli && \
    rm /tmp/signal-cli.tar.gz || true

# Create application directory
WORKDIR /app

# Copy requirements first for better caching
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application files
COPY radical_sea_bunny.py .
COPY config/ ./config/
COPY scripts/ ./scripts/
COPY docker/entrypoint.sh /entrypoint.sh

# Create necessary directories
RUN mkdir -p /app/.radical_sea_bunny \
    /app/.radical_sea_bunny/payloads \
    /app/.radical_sea_bunny/workspaces \
    /app/.radical_sea_bunny/scans \
    /app/.radical_sea_bunny/reports \
    /app/.radical_sea_bunny/phishing_pages \
    /app/.radical_sea_bunny/phishing_templates \
    /app/.radical_sea_bunny/captured_credentials \
    /app/.radical_sea_bunny/ssh_keys \
    /app/.radical_sea_bunny/traffic_logs \
    /app/.radical_sea_bunny/nikto_results \
    /app/.radical_sea_bunny/web_templates \
    /app/.radical_sea_bunny/sessions \
    /app/radical_sea_bunny_reports \
    /app/radical_sea_bunny_reports/graphics \
    /app/temp

# Set permissions
RUN chmod +x /entrypoint.sh

# Expose ports
EXPOSE 5000 8080 22

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD python -c "import socket; socket.create_connection(('localhost', 5000), timeout=5)" || exit 1

# Entry point
ENTRYPOINT ["/entrypoint.sh"]

# Default command
CMD ["python", "radical_sea_bunny.py"]
